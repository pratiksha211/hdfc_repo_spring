create database movieDB;

use movieDB;

CREATE TABLE Movies(movie_id int primary key, movie_title varchar(200), genre varchar(30), director varchar(50) , duration int , rating float , release_date date);

CREATE TABLE Theatres( theatre_id int primary key ,theatre_name varchar(200),location varchar(50) , seating_capacity int);

CREATE TABLE Shows (show_id int primary key, theatre_id int, movie_id int, show_date date, show_time time, available_seats int, foreign key (theatre_id) references Theatres(theatre_id), foreign key (movie_id) references Movies(movie_id));

CREATE TABLE Bookings(booking_id int primary key, customer_id int, show_id int, booking_date date, total_amount float ,payment_method varchar(50),foreign key (show_id) references Shows(show_id));
